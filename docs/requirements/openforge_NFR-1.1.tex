%% 1. INTRODUCTION
\section{Introduction}

\subsection{Purpose}
This document specifies the non-functional requirements (NFRs) for OpenForge. Where the Functional Requirements document (FR-1.1) defines \emph{what} the system does, this document defines \emph{how well} it does it --- the quality attributes, constraints, and system-wide properties governing the platform's architecture and implementation.

A key architectural principle running through all NFRs is that OpenForge shall remain \textbf{lightweight}: it maximises use of GitHub's existing infrastructure and avoids building anything the platform does not uniquely need to own. This principle directly shapes performance, maintainability, and portability requirements throughout this document.

NFRs are organised following the ISO/IEC 25010 quality model.

\subsection{Requirement Identifier Convention}
Each requirement is assigned a unique identifier of the form \texttt{[NFR-XX-NN]}, where \texttt{XX} denotes the quality category and \texttt{NN} is a sequential number.

%% 5. SECURITY
\section{Security}

\subsection{Authentication and Authorisation}
\begin{longtable}{L{2.2cm} L{1.8cm} L{9.3cm}}
  \rowcolor{ofaccent}\color{white}\bfseries ID & \color{white}\bfseries Priority & \color{white}\bfseries Requirement \\
  \endfirsthead
  \rowcolor{ofaccent}\color{white}\bfseries ID & \color{white}\bfseries Priority & \color{white}\bfseries Requirement \\
  \endhead\hline
  \texttt{NFR-SE-01} & Must   & All user authentication shall be performed via \textbf{GitHub OAuth 2.0}. The platform shall not store user passwords. \\
  \rowcolor{ofrowalt}
  \texttt{NFR-SE-02} & Must   & Session tokens shall expire after \textbf{30 days} of inactivity. Users shall be re-authenticated via GitHub OAuth on expiry. \\
  \texttt{NFR-SE-03} & Must   & All API endpoints that write data shall require a valid authenticated session. Read-only public endpoints may be unauthenticated. \\
  \rowcolor{ofrowalt}
  \texttt{NFR-SE-04} & Must   & GitHub webhook payloads shall be verified using the \textbf{HMAC-SHA256 signature} in the \texttt{X-Hub-Signature-256} header. Payloads failing verification shall be rejected and logged. \\
  \texttt{NFR-SE-05} & Must   & The CODEOWNERS file generation step (maintainer nomination) shall only be performed using a GitHub token with the minimum required repository scope. The token shall not be stored in platform logs. \\
  \hline
\end{longtable}

\subsection{Data Protection}
\begin{longtable}{L{2.2cm} L{1.8cm} L{9.3cm}}
  \rowcolor{ofaccent}\color{white}\bfseries ID & \color{white}\bfseries Priority & \color{white}\bfseries Requirement \\
  \endfirsthead
  \rowcolor{ofaccent}\color{white}\bfseries ID & \color{white}\bfseries Priority & \color{white}\bfseries Requirement \\
  \endhead\hline
  \texttt{NFR-SE-06} & Must   & All data in transit shall be encrypted using \textbf{TLS 1.2 or higher}. TLS 1.0 and 1.1 shall be disabled. \\
  \rowcolor{ofrowalt}
  \texttt{NFR-SE-07} & Must   & User email addresses obtained via GitHub OAuth shall never be stored in plain text in application logs or displayed publicly. \\
  \texttt{NFR-SE-08} & Should & The platform shall implement \textbf{Content Security Policy (CSP)} headers to mitigate XSS attacks. \\
  \rowcolor{ofrowalt}
  \texttt{NFR-SE-09} & Must   & All user-submitted content (request descriptions, review comments) shall be sanitised to prevent stored XSS before rendering. \\
  \hline
\end{longtable}

\subsection{Rate Limiting and Abuse Prevention}
\begin{longtable}{L{2.2cm} L{1.8cm} L{9.3cm}}
  \rowcolor{ofaccent}\color{white}\bfseries ID & \color{white}\bfseries Priority & \color{white}\bfseries Requirement \\
  \endfirsthead
  \rowcolor{ofaccent}\color{white}\bfseries ID & \color{white}\bfseries Priority & \color{white}\bfseries Requirement \\
  \endhead\hline
  \texttt{NFR-SE-10} & Must   & Upvote and annotation actions shall be rate-limited per authenticated user. Default limit: 60 voting actions per hour per account. \\
  \rowcolor{ofrowalt}
  \texttt{NFR-SE-11} & Must   & A single account shall not cast more than \textbf{one upvote per request}. Duplicate attempts shall be rejected silently. \\
  \texttt{NFR-SE-12} & Should & Automated bot detection shall be applied to account creation and high-frequency voting patterns, with flagged accounts entering a manual review queue. \\
  \rowcolor{ofrowalt}
  \texttt{NFR-SE-13} & Must   & All moderation actions shall be logged with timestamp, actor ID, and action type in an append-only audit log inaccessible to non-moderators. \\
  \hline
\end{longtable}
